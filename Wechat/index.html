/** * get请求 * @param string $url 请求的地址 * @return array 请求到的结果 */ function get($url){ $ch = curl_init(); curl_setopt($ch, CURLOPT_URL, $url); curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1); //以结果的方式返回你请求到的数据 $res = curl_exec($ch); curl_close(); $arr = json_decode($res, true); return $arr; }
